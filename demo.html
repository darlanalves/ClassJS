Class declaration

Whapp.define('MyClass', { foo: function() { alert('baz'); } }); var test = new MyClass(); test.foo(); // alerts 'baz'

Class declaration (with namespace)

Whapp.define('My.other.Class', { foo: function() { alert('other!'); } }); var test = new My.other.Class(); test.foo(); // alerts 'other!'

Statics

Whapp.define('My.other.ClassWithStatics', { consts: [ 'MY_CONST', 'OTHER_CONST' ], statics: { foo: function() { alert(new Date); } } }); My.other.ClassWithStatics.foo(); // alerts a Date object alert(My.other.ClassWithStatics.MY_CONST);
